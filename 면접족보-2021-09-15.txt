import java.util.Scanner;


class Calculate{
	
	private int num1;
	private int num2;
	private String mathsign;
	private String check;
	
	Scanner sc = new Scanner(System.in);
	
	public void MathSign() {
		
		if(mathsign.equals("+")) {
			System.out.println(num1 + num2);
		}
		
		else if(mathsign.equals("-")) {
			System.out.println(num1 - num2);
		}
		
		else if(mathsign.equals("*")) {
			System.out.println(num1 * num2);
		}
		
		else if(mathsign.equals("/")) {
			System.out.println(num1 / num2);
		}
	}
	
	public void show() {
		while(true) {
			System.out.println("첫 번째 숫자를 입력하세요.");
			num1 = sc.nextInt();
			
			System.out.println("연산 기호를 입력하세요.");
			mathsign = sc.next();		
			
			System.out.println("두 번째 숫자를 입력하세요.");
			num2 = sc.nextInt();	
			
			MathSign();
			System.out.println("종료하시겠습니까? 종료:y 계속:n");
			check = sc.next();
			
			if(check.equals("y") || check.equals("Y")) {
				System.out.println("종료입니다.");
				break;
			}
			else {
				continue;
			}
		}
	}
	
}

소스코드

import java.util.Scanner;



class Random{
	
	private int random = (int) (Math.random() * 100) + 1;
	private int num;
	private int count = 10;
	
	Scanner sc = new Scanner(System.in);
	
	public void show() {
		System.out.println("1 ~ 100 사이의 숫자를 입력하세요");
		System.out.println("남은 횟수 : " + count);
		while(true) {
			num = sc.nextInt();
			if(count > 0) {
				if(random > num) {
					System.out.println("UP");
					System.out.println("남은 횟수 : " + count);
					count--;
					continue;
				}
				else if(random < num) {
					System.out.println("DOWN");
					System.out.println("남은 횟수 : " + count);
					count--;
					continue;
				}
				else {
					System.out.println("축하드립니다 정답입니다. " + count + "회 만에 맞추셨습니다");
					System.out.println("종료하겠습니다.");
					break;
				}
			}
			else {
				System.out.println("게임에서 졌기 때문에 종료합니다");
				break;
			}
		}
	}
}


public class LastExam01 {

	public static void main(String[] args) {
		// TODO Auto-generated method stub
		Random random = new Random();
		random.show();
	}

}
